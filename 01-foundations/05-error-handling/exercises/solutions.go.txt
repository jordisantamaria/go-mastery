package exercises

import (
	"fmt"
	"strconv"
	"strings"
)

func SafeDivide(a, b float64) (float64, error) {
	if b == 0 {
		return 0, ErrDivisionByZero
	}
	return a / b, nil
}

func Sqrt(n float64) (float64, error) {
	if n < 0 {
		return 0, ErrNegativeNumber
	}
	if n == 0 {
		return 0, nil
	}
	guess := n / 2
	for range 10 {
		guess = (guess + n/guess) / 2
	}
	return guess, nil
}

func ParseAge(s string) (int, error) {
	if s == "" {
		return 0, fmt.Errorf("parse age: %w", ErrEmpty)
	}
	n, err := strconv.Atoi(s)
	if err != nil {
		return 0, fmt.Errorf("parse age %q: %w", s, err)
	}
	if n < 0 {
		return 0, fmt.Errorf("parse age: %d: %w", n, ErrNegativeNumber)
	}
	if n > 150 {
		return 0, fmt.Errorf("parse age: %d: %w", n, ErrOutOfRange)
	}
	return n, nil
}

func SafeGet(items []string, index int) (string, error) {
	if len(items) == 0 {
		return "", ErrEmpty
	}
	if index < 0 || index >= len(items) {
		return "", ErrOutOfRange
	}
	return items[index], nil
}

func FindUser(users []User, name string) (*User, error) {
	if name == "" {
		return nil, fmt.Errorf("FindUser: %w", ErrEmpty)
	}
	for i := range users {
		if users[i].Name == name {
			return &users[i], nil
		}
	}
	return nil, fmt.Errorf("FindUser: %q: %w", name, ErrNotFound)
}

type FieldError struct {
	Field   string
	Message string
}

func (e *FieldError) Error() string {
	return fmt.Sprintf("field %s: %s", e.Field, e.Message)
}

func ValidateUser(name, email string, age int) error {
	if name == "" {
		return &FieldError{Field: "name", Message: "is required"}
	}
	if !strings.Contains(email, "@") {
		return &FieldError{Field: "email", Message: "must contain @"}
	}
	if age < 0 || age > 150 {
		return &FieldError{Field: "age", Message: "must be between 0 and 150"}
	}
	return nil
}

func ValidateAll(name, email string, age int) error {
	var errs []string

	if name == "" {
		errs = append(errs, "name: required")
	}
	if !strings.Contains(email, "@") {
		errs = append(errs, "email: must contain @")
	}
	if age < 0 || age > 150 {
		errs = append(errs, "age: must be between 0 and 150")
	}

	if len(errs) > 0 {
		return fmt.Errorf(strings.Join(errs, "; "))
	}
	return nil
}
